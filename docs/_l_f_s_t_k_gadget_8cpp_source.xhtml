419 void LFSTK_gadgetClass::drawLabel(gadgetStruct* details)
420 {
421  int labelx=0;
422  int labely;
423  cairoColor lcol;
424  cairoColor *colptr=&this->labelBGColour.RGBAColour;
425  geometryStruct labelrect;
426  int labh;
427  int laby;
428 
429  if(this->isActive==false)
430  details->state=INACTIVECOLOUR;
431 
432  labely=(details->gadgetGeom.h/2)+(0.5-this->fontExtents.descent+this->fontExtents.height/2);
433 
434  if(strcmp(this->label,"--")!=0)
435  {
436  switch(this->labelGravity)
437  {
438  case MENU:
439  labelx=details->gadgetGeom.h+(this->pad*4);
440  break;
441  case LEFT:
442  if(this->useImage==true)
443  labelx=details->reserveSpace+pad*2;
444  else
445  labelx=details->reserveSpace+(this->pad*2);
446  break;
447  case CENTRE:
448  if(this->useImage==true)
449  {
450  if(this->imageGravity==RIGHT)
451  labelx=((details->gadgetGeom.w-this->imageWidth)/2)-(this->textExtents.width/2);
452  else

454  }

457 
458  if(details->hasIndicator==true)
459  labelx-=(details->indicatorGeom.w/2);
460  break;
461  case RIGHT:
462  if(this->useImage==true)
463  labelx=details->gadgetGeom.w-(this->textExtents.width)-pad*2;
464  else
465  labelx=details->gadgetGeom.w-this->textExtents.width-this->pad*2;

467  case TOOLBAR:
468  labelx=((details->gadgetGeom.w)/2)-(this->textExtents.width/2);

498  lcol.g=1;

501  }
502  colptr=&lcol;
503  }
504 
505  cairo_set_source_rgba(this->cr,colptr->r,colptr->g,colptr->b,colptr->a);
506  cairo_rectangle(this->cr,labelrect.x,labelrect.y,labelrect.w,labelrect.h);

508  }
509 
510  cairo_move_to(this->cr,labelx,labely);
511  cairo_set_source_rgba(this->cr,this->fontColourNames[details->state].RGBAColour.r,this->fontColourNames[details->state].RGBAColour.g,this->fontColourNames[details->state].RGBAColour.b,1.0);
