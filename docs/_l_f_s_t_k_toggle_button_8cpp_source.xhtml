197 bool LFSTK_toggleButtonClass::mouseUp(XButtonEvent *e)
198 {
199  gadgetState col=NORMALCOLOUR;
200 
201  if(this->inWindow==true)
202  {
203  this->toggleState=!this->toggleState;
204  col=PRELIGHTCOLOUR;
205  }
206 
207  if(this->boxStyle==TOGGLECHECK)
208  {
209  if(this->toggleState==true)
210  {
211  if(this->useImage==true)
212  this->cImage=this->checkOn;
213  this->gadgetDetails.state=ACTIVECOLOUR;
214  }
215  else
216  {
217  if(this->useImage==true)
218  this->cImage=this->checkOff;
219  this->gadgetDetails.state=NORMALCOLOUR;
220  }
221  }
222 
223  LFSTK_gadgetClass::LFSTK_clearWindow();
224  XSync(this->display,false);
225 
226  if(this->inWindow==true)
227  {
228  if(this->callback.releaseCallback!=NULL)
229  return(this->callback.releaseCallback(this,this->callback.userData));
230  }
231  return(true);
232 }
