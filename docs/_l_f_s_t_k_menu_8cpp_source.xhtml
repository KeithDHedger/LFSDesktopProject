66 void LFSTK_menuClass::LFSTK_showMenu(void)
67 {
68  XEvent event;
69  Window dw;
70  int x;
71  int y;
72  unsigned int w,h,dump;
73  Window sink;
74  Window childwindow;
75  int sinkx;
76  int sinky;
77  unsigned int buttonmask;
78 
79  this->mainMenuWindow->LFSTK_showWindow(true);
80  this->mainLoop=true;
81 
82  XTranslateCoordinates(this->display,this->parentwc->window,this->parentwc->rootWindow,this->x,this->y,&x,&y,&dw);
83  this->mainMenuWindow->LFSTK_moveWindow(x,y,true);
84  XSetInputFocus(this->mainMenuWindow->display,this->mainMenuWindow->window,RevertToNone,CurrentTime);
85  XSync(display,false);
86 
87  while(XPending(this->display))
88  {
89  XNextEvent(this->mainMenuWindow->display,&event);
90  mappedListener *ml=this->mainMenuWindow->LFSTK_getMappedListener(event.xany.window);
91 
92  if(ml!=NULL)
93  ml->function(ml->gadget,&event,ml->type);
94  }
95 
96  while(this->mainLoop==true)
97  {
98  if(XPending(this->display))
99  {
100  XNextEvent(this->mainMenuWindow->display,&event);
101  }
102  else
103  {
104  XQueryPointer(this->display,this->parentwc->rootWindow,&sink,&childwindow,&sinkx,&sinky,&sinkx,&sinky,&buttonmask);
105  if((childwindow!=this->mainMenuWindow->window) && (buttonmask!=0))
106  {
107  this->mainMenuWindow->LFSTK_hideWindow();
108  this->mainMenuWindow->mainLoop=false;
109  this->mainLoop=false;
110  }
111  continue;
112  }
113 
114  mappedListener *ml=this->mainMenuWindow->LFSTK_getMappedListener(event.xany.window);
115 
116  if(ml!=NULL)
117  ml->function(ml->gadget,&event,ml->type);
118 
119  if(this->mainMenuWindow->LFSTK_handleWindowEvents(&event)<0)
120  mainLoop=false;
121 
122  switch(event.type)
123  {
124  case ConfigureNotify:
125  case Expose:
126  XTranslateCoordinates(this->display,this->parentwc->window,this->parentwc->rootWindow,this->x,this->y,&x,&y,&dw );
127  this->mainMenuWindow->LFSTK_moveWindow(x,y,true);
128  break;
129 
130  case FocusOut:
131  case FocusIn:
132  this->mainMenuWindow->LFSTK_hideWindow();
133  this->mainMenuWindow->mainLoop=false;
134  this->mainLoop=false;
135  break;
136  }
137  }
138 }

146 void LFSTK_menuClass::LFSTK_addMainMenus(menuStruct **menus,int menucnt)
147 {
148  LFSTK_menuItemClass *label;
149  int sy=0;
150  double txtwid=0;
151  double maxtxtwid=0;
152  int finaltxtwid;
153  int gotsubmenu=8;
154  int gotthumb=LEFT;
155  int winshrink=0;
156  int hite=GADGETHITE;
157 
158  if(this->mainMenu!=NULL)
159  {
160  for(int j=0;j<this->subwindows->size();j++)
161  delete this->subwindows->at(j);
162  this->subwindows->clear();
163  LFSTK_freeMenus(this->mainMenu,this->mainMenuCnt);
164  }
165 
166  this->mainMenuCnt=menucnt;
167  this->mainMenu=menus;
168 
169  for(int j=0;j<menucnt;j++)
170  {
171  txtwid=this->LFSTK_getTextWidthForFont(this->mainMenu[j]->label);
172  if(txtwid>maxtxtwid)
173  maxtxtwid=txtwid;
174  if(this->mainMenu[j]->hasSubMenu==true)
175  gotsubmenu=GADGETHITE;
176  if(this->mainMenu[j]->imageType!=NOTHUMB)
177  gotthumb=MENU;
178  if(strcmp(this->mainMenu[j]->label,"--")==0)
179  winshrink+=(GADGETHITE-SEPARATORHITE);
180  }
181 
182  if(gotthumb==MENU)
183  maxtxtwid+=GADGETHITE+gotsubmenu;
184  else
185  maxtxtwid+=gotsubmenu;
186 
187  this->mainMenuWindow=new LFSTK_toolWindowClass(this->display,this->parentwc,"_NET_WM_WINDOW_TYPE_MENU",this->x,this->y,maxtxtwid,GADGETHITE*this->mainMenuCnt-winshrink,"menu window");
188  for(int j=0;j<this->mainMenuCnt;j++)
189  {
190  hite=GADGETHITE;
191  if(strcmp(this->mainMenu[j]->label,"--")==0)
192  hite=SEPARATORHITE;
193 
194  label=new LFSTK_menuItemClass(this->mainMenuWindow,this,0,sy,maxtxtwid,hite,this->mainMenu[j],gotthumb);
195  label->LFSTK_setCallBack(this->callback.pressCallback,this->callback.releaseCallback,(void*)(this->mainMenu[j]->userData));
196  if(this->mainMenu[j]->imageType==FILETHUMB)
197  label->LFSTK_setImageFromPath(this->mainMenu[j]->data.imagePath,MENU,true);
198  if(this->mainMenu[j]->imageType==CAIROTHUMB)
199  label->LFSTK_setImageFromSurface(this->mainMenu[j]->data.surface,MENU,true);
200  sy+=hite;
201  }
202 
203  this->subwindows->push_back(this->mainMenuWindow);
204 }

230 int LFSTK_menuClass::LFSTK_getTextWidthForFont(const char *text)
231 {
232  cairo_text_extents_t returnextents;
233  cairo_t *cr=NULL;
234  cairo_surface_t *sfc=NULL;
235  char *string=strdup(this->fontDesc);
236  char *str;
237  bool found=false;
238 
239  weight=CAIRO_FONT_WEIGHT_NORMAL;
240  slant=CAIRO_FONT_SLANT_NORMAL;
241  fontSize=10;
242 
243  if(this->fontName!=NULL)
244  free(this->fontName);
245 
246  this->fontName=strdup("sans");
247  str=strtok(string,":");
248  while(1)
249  {
250  found=false;
251  if(str==NULL)
252  break;
253  if(strcasecmp(str,"bold")==0)
254  {
255  this->weight=CAIRO_FONT_WEIGHT_BOLD;
256  found=true;
257  }
258  if(strcasecmp(str,"italic")==0)
259  {
260  this->slant=CAIRO_FONT_SLANT_ITALIC;
261  found=true;
262  }
263  if(strcasestr(str,"size=")!=NULL)
264  {
265  this->fontSize=atoi(&str[5]);
266  found=true;
267  }
268 
269  if(found==false)
270  {
271  if(this->fontName!=NULL)
272  free(this->fontName);
273  this->fontName=strdup(str);
274  }
275  str=strtok(NULL,":");
276  }
277  free(string);
278 
279  this->parentwc->globalLib->LFSTK_setCairoSurface(this->parentwc->display,this->parentwc->window,this->parentwc->visual,&sfc,&cr,1,1);
280  cairo_select_font_face(cr,this->fontName,this->slant,this->weight);
281  cairo_set_font_size(cr,this->fontSize);
282  cairo_text_extents(cr,text,&returnextents);
283 
284  cairo_destroy(cr);
285  cairo_surface_destroy(sfc);
286  return((int)(returnextents.x_advance+0.5));
287 
288 }
