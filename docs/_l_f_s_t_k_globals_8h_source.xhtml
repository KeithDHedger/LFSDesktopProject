77 static bool showFileData=true;
78 static void debugFunc(const char *file,const char *func,int line,const char *fmt, ...)
79 {
80  va_list ap;
81  char *buffer,*subbuffer;
82 
83  buffer=(char*)alloca(512);
84  subbuffer=(char*)alloca(512);
85 
86  buffer[0]=0;
87  subbuffer[0]=0;
88  va_start(ap, fmt);
89  while (*fmt)
90  {
91  subbuffer[0]=0;
92  if(fmt[0]=='%')
93  {
94  fmt++;
95  switch(*fmt)
96  {
97  case 's':
98  sprintf(subbuffer,"%s",va_arg(ap,char*));
99  break;
100  case 'i':
101  sprintf(subbuffer,"%i",va_arg(ap,int));
102  break;
103  case 'p':
104  sprintf(subbuffer,"%p",va_arg(ap,char*));
105  break;
106  case 'c':
107  sprintf(subbuffer,"%c",va_arg(ap,int));
108  break;
109  case 'f':
110  sprintf(subbuffer,"%f",va_arg(ap,double));
111  break;
112  default:
113  sprintf(subbuffer,"%c",fmt[0]);
114  break;
115  }
116  }
117  else
118  sprintf(subbuffer,"%c",fmt[0]);
119  strcat(buffer,subbuffer);
120  fmt++;
121  }
122  va_end(ap);
123 
124  if(showFileData==true)
125  {
126  printf("\nFile: %s\nFunc: %s\nLine: %i\n",basename(file),func,line);
127  printf("----USER DATA----\n%s\n----END----\n",buffer);
128  }
129  else
130  {
131  printf("%s\n",buffer);
132  }
133 }
134 #define DEBUGFUNC(x,...) debugFunc(__FILE__,__func__,__LINE__,(const char*)x,__VA_ARGS__)
135 #else
136 #define DEBUGFUNC(...) printf("Remove debug code here: %s:%i\n",__FILE__,__LINE__);
137 #endif
