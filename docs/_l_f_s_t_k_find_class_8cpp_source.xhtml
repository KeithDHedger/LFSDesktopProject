222 static bool sortDataTN(dataStruct i,dataStruct j)
223 {
224  if(fc->LFSTK_getSort()==true)

226  if(i.fileType<j.fileType)

236  return(true);
237 
238  if((i.fileType==j.fileType) && (i.name<j.name))
239  return(true);

241  }
242  return(false);
243 }

304 bool LFSTK_findClass::fileTypeTest(int filetype)
305 {
306  if(this->LFSTK_getFindType()==ANYTYPE)
307  return(true);
308 
309  switch(this->LFSTK_getFindType())

311  case FOLDERTYPE:

313  return(true);
314  break;
315 
316  case FILETYPE:
317  if((filetype==FILETYPE) || (filetype==FILELINKTYPE))
318  return(true);
319  break;
320  }
321  return(false);

348 void LFSTK_findClass::LFSTK_findFiles(const char *dir,bool multi)
349 {
350  DIR *dirhandle;
351  dirent *entry;
352  struct stat filestat;
353  int retstat;
354  dataStruct datas;
355  char *filepath;
356 
357  fc=this;
358  if(multi==false)
359  this->deleteData();
360  filepath=(char*)alloca(PATH_MAX);
361  dirhandle=opendir(dir);
362  if(dirhandle!=NULL)
363  {
364  while ((entry=readdir(dirhandle)) != NULL)
365  {
366  if(strcmp(entry->d_name,".")==0)
367  continue;
368  if((this->LFSTK_getIncludeHidden()==false) && ((strlen(entry->d_name)>2) && ((entry->d_name[0]=='.') && (entry->d_name[1]!='.') )))
369  continue;
370  if((this->LFSTK_getFileTypes()!=NULL) && (strcasestr(entry->d_name,this->LFSTK_getFileTypes())==NULL))
371  continue;
372  if((this->ignoreNavLinks==true) && ((strcmp(entry->d_name,".")==0) || (strcmp(entry->d_name,"..")==0)))
373  continue;
374 
375  datas.name=entry->d_name;
376  sprintf(filepath,"%s/%s",dir,entry->d_name);
377  if(entry->d_type==DT_LNK)
378  {
379  retstat=stat(filepath,&filestat);
380  if(retstat!=0)
381  {
382  if(this->LFSTK_getIgnoreBroken()==true)
383  continue;
384  datas.fileType=BROKENLINKTYPE;
385  }
386  else
387  {
388  if(this->LFSTK_getFollowlinks()==false)
389  {
390  datas.fileType=FILELINKTYPE;
391  }
392  else
393  {
394  switch(filestat.st_mode & S_IFMT)
395  {
396  case S_IFREG:
397  if(this->fileTypeTest(FILELINKTYPE)==false)
398  continue;
399  datas.fileType=FILELINKTYPE;
400  break;
401  case S_IFDIR:
402  if(this->fileTypeTest(FOLDERLINKTYPE)==false)
403  continue;
404  datas.fileType=FOLDERLINKTYPE;
405  break;
406  }
407  }
408  }
409  }
410  else
411  {
412  switch(entry->d_type)
413  {
414  case DT_REG:
415  if(this->fileTypeTest(FILETYPE)==false)
416  continue;
417  datas.fileType=FILETYPE;
418  break;
419  case DT_DIR:
420  if(this->fileTypeTest(FOLDERTYPE)==false)
421  continue;
422  datas.fileType=FOLDERTYPE;
423  break;
424  }
425  }
426  if(this->LFSTK_getFullPath()==true)
427  datas.path=filepath;
428  else
429  datas.path="";
430 
431  this->data.push_back(datas);
432  datas.name.clear();
433  datas.path.clear();
434  }
435  closedir(dirhandle);
436  }
437  this->dataCnt=this->data.size();
438 }
