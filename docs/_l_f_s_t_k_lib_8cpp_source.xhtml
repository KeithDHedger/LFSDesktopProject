389 bool LFSTK_lib::LFSTK_loadVarsFromFile(const char* filepath,const args* dataptr)
390 {
391  FILE* fd=NULL;
392  char buffer[2048];
393  int cnt;
394  char* argname=NULL;
395  char* strarg=NULL;
396 
397  fd=fopen(filepath,"r");
398  if(fd!=NULL)
399  {
400  while(feof(fd)==0)
401  {
402  buffer[0]=0;
403  fgets(buffer,2048,fd);
404  argname=NULL;
405  strarg=NULL;
406  sscanf(buffer,"%ms %ms",&argname,&strarg);
407  cnt=0;
408  while(dataptr[cnt].name!=NULL)
409  {
410  if((strarg!=NULL) && (argname!=NULL) && (strcmp(argname,dataptr[cnt].name)==0))
411  {
412  switch(dataptr[cnt].type)
413  {
414  case TYPEINT:
415  *(int*)dataptr[cnt].data=atoi(strarg);
416  break;
417  case TYPESTRING:
418  if(*(char**)(dataptr[cnt].data)!=NULL)
419  free(*(char**)(dataptr[cnt].data));
420  sscanf(buffer,"%*s %m[^\n]s",(char**)dataptr[cnt].data);
421  break;
422  case TYPEBOOL:
423  *(bool*)dataptr[cnt].data=(bool)atoi(strarg);
424  break;
425  }
426  }
427  cnt++;
428  }
429  if(argname!=NULL)
430  free(argname);
431  if(strarg!=NULL)
432  free(strarg);
433  }
434  fclose(fd);
435  return(true);
436  }
437  return(false);
438 }

559  gadget->isDoubleClick=true;

750 char* LFSTK_lib::LFSTK_oneLiner(const char* fmt,...)
751 {
752  FILE *fp;
753  va_list ap;
754  char *buffer,*subbuffer;
755 
756  buffer=(char*)alloca(MAXBUFFER);
757  subbuffer=(char*)alloca(MAXBUFFER);
758 
759  buffer[0]=0;
760  subbuffer[0]=0;
761  va_start(ap, fmt);
762  while (*fmt)
763  {
764  subbuffer[0]=0;
765  if(fmt[0]=='%')
766  {
767  fmt++;
768  switch(*fmt)
769  {
770  case 's':
771  sprintf(subbuffer,"%s",va_arg(ap,char*));
772  break;
773  case 'i':
774  sprintf(subbuffer,"%i",va_arg(ap,int));
775  break;
776  case '%':
777  sprintf(subbuffer,"%%");
778  break;
779  default:
780  sprintf(subbuffer,"%c",fmt[0]);
781  break;

920 cairo_surface_t* LFSTK_lib::cairo_image_surface_create_from_jpeg_mem(const unsigned char* data, size_t len)
921 {
922  struct jpeg_decompress_struct cinfo;
923  struct jpeg_error_mgr jerr;
924  JSAMPROW row_pointer[1];
925  cairo_surface_t *sfc;
926 
927  // initialize jpeg decompression structures
928  cinfo.err=jpeg_std_error(&jerr);
929  jpeg_create_decompress(&cinfo);
930  jpeg_mem_src(&cinfo,(const unsigned char*)data, len);
931  jpeg_read_header(&cinfo,true);
932 
933 #if __BYTE_ORDER__ == __ORDER_LITTLE_ENDIAN__
934  cinfo.out_color_space=JCS_EXT_BGRA;
935 #else
936  cinfo.out_color_space=JCS_EXT_ARGB;
937 #endif
938 
939  // start decompressor
940  jpeg_start_decompress(&cinfo);
941 
942  // create Cairo image surface
943  sfc=cairo_image_surface_create(CAIRO_FORMAT_RGB24,cinfo.output_width,cinfo.output_height);
944  if(cairo_surface_status(sfc)!=CAIRO_STATUS_SUCCESS)
945  {
946  jpeg_destroy_decompress(&cinfo);
947  return(sfc);
948  }
949 
950  // loop over all scanlines and fill Cairo image surface
951  while(cinfo.output_scanline<cinfo.output_height)
952  {
953  row_pointer[0]=cairo_image_surface_get_data(sfc)+(cinfo.output_scanline * cairo_image_surface_get_stride(sfc));
954  jpeg_read_scanlines(&cinfo,row_pointer,1);
955  }
956 
957  // finish and close everything
958  cairo_surface_mark_dirty(sfc);
959  jpeg_finish_decompress(&cinfo);
960  jpeg_destroy_decompress(&cinfo);
961 
962  // set jpeg mime data
963  cairo_surface_set_mime_data(sfc,CAIRO_MIME_TYPE_JPEG,(const unsigned char*)data,len,free,(void*)data);
964 
965  return sfc;
966 }

1000  return(NULL);

1005  if(ptr[j]!=magic[j])
1006  flag=false;
1007 
1008  close(infile);
1009  if(flag==false)
1010  {

1012  return(NULL);
