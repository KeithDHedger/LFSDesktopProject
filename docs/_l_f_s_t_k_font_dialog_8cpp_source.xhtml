41 void LFSTK_fontDialogClass::buildFontString(void)
42 {
43  const char *boldstr="";
44  const char *italicstr="";
45  const char *dobold="b";
46  const char *doitalic="i";
47  char *formatstring=NULL;
48  this->fontData.bold=false;
49  this->fontData.italic=false;
50 
51  if(this->boldcheck->LFSTK_getValue()==true)
52  {
53  boldstr=":bold";
54  dobold="B";
55  this->fontData.bold=true;
56  }
57  if(this->italiccheck->LFSTK_getValue()==true)
58  {
59  italicstr=":italic";
60  doitalic="I";
61  this->fontData.italic=true;
62  }
63  this->selectedFontNumber=fontlist->LFSTK_getCurrentListItem();
64 
65  if(this->fontData.fontString!=NULL)
66  free(this->fontData.fontString);
67  asprintf(&this->fontData.fontString,"%s:size=%s%s%s",this->fontsAZ[this->selectedFontNumber],this->fontsize->LFSTK_getCStr(),boldstr,italicstr);
68  asprintf(&formatstring,"ns%s%s",dobold,doitalic);
69  this->preview->LFSTK_setCairoFontDataParts(formatstring,this->fontsAZ[selectedFontNumber],atoi(this->fontsize->LFSTK_getCStr()));
70  this->preview->LFSTK_clearWindow();
71 
72  this->fontData.fontString;
73  this->fontData.fontName=this->fontsAZ[selectedFontNumber];
74  this->fontData.isValid=true;
75  free(formatstring);
76  this->preview->LFSTK_clearWindow();
77 }

112 void LFSTK_fontDialogClass::loadFontStrings(void)
113 {
114  unsigned fontcnt=0;
115  FILE *fp;
116  char *command;
117  char *out=NULL;
118  char line[1024];
119 
120  line[0]=0;
121 
122  asprintf(&command,"%s","fc-list : family|awk -F, '{print $1}'|sort -u|wc -l");
123  out=wc->globalLib->LFSTK_oneLiner("%s",command);
124  if(out==NULL)
125  fontcnt=0;
126  else
127  {
128  fontcnt=atoi(out);
129  free(out);
130  }
131  free(command);
132 
133  fontsAZ=new char*[fontcnt];
134  maxFonts=fontcnt;
135 
136  asprintf(&command,"%s","fc-list : family|awk -F, '{print $1}'|sort -ur");
137  fp=popen(command, "r");
138  if(fp!=NULL)
139  {
140  while(fgets(line,1024,fp))
141  {
142  fontcnt--;
143  line[strlen(line)-1]=0;
144  fontsAZ[fontcnt]=strdup(line);
145  }
146  pclose(fp);
147  }
148  free(command);
149 }

151 bool LFSTK_fontDialogClass::LFSTK_showDialog(const char* font)
152 {
153  bool mainLoop=true;
154  XEvent event;
155  geometryStruct geomfont;
156  pointStruct pt;
157 
158  this->dialog->LFSTK_showWindow(true);
159  this->dialog->LFSTK_setKeepAbove(true);
160  this->dialog->LFSTK_setTransientFor(this->wc->window);
161  this->mainLoop=true;
162 
163 //TODO//
164  //this->fontsize->LFSTK_setFocus();
165  while(this->mainLoop==true)
166  {
167  XNextEvent(this->dialog->display,&event);
168  mappedListener *ml=this->dialog->LFSTK_getMappedListener(event.xany.window);
169  if(ml!=NULL)
170  ml->function(ml->gadget,&event,ml->type);
171 
172 
173  if(this->dialog->LFSTK_handleWindowEvents(&event)<0)
174  mainLoop=false;
175 
176  if(ml==NULL)
177  continue;
178 
179  switch(event.type)
180  {
181  case ButtonRelease:
182  if(ml!=NULL)
183  {
184  if(ml->gadget==this->apply)
185  {
186  this->LFSTK_getFontData(true);
187  this->fontData.isValid=true;
188  this->mainLoop=false;
189  }
190  if(ml->gadget==this->cancel)
191  {
192  this->mainLoop=false;
193  this->fontData.isValid=false;
194  }
195  }
196  break;
197  }
198  }
199  this->dialog->LFSTK_hideWindow();
200  return(true);
201 }
