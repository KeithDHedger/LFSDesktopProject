246 void LFSTK_multiLineEditClass::drawText(void)
247 {
248  int cursorwidth;
249  int startchar=0;
250  int len=this->cursorPos;
251  char *buffer;
252  const char *curs="";
253  double yoffset=0;
254  cairo_text_extents_t partextents;
255  cairo_text_extents_t charextents;
256 
257  cairo_save(this->cr);
258  cairo_reset_clip(this->cr);
259  cairo_set_source_rgba(this->cr,1.0,1.0,1.0,1.0);
260  cairo_paint(this->cr);
261  cairo_restore(this->cr);
262 
263  cairo_save(this->cr);
264  cairo_select_font_face(this->cr,fontName,slant,weight);
265  cairo_set_font_size(this->cr,fontSize);
266  cairo_set_source_rgba(this->cr,0.0,0,0,1.0);
267  for (int j=0;j<this->lines.size();j++)
268  {
269  yoffset+=this->textExtents.height;
270  cairo_move_to(this->cr,this->pad,yoffset);
271 //do cursor
272  if(lines.at(j)->cursorPos!=-1)
273  {
274  char *data;
275  char undercurs[2];
276  char *aftercursor;
277 
278  asprintf(&data,"%s",lines.at(j)->line);
279  undercurs[0]=data[lines.at(j)->cursorPos];
280  undercurs[1]=0;
281  if(undercurs[0]==0)
282  undercurs[0]=' ';
283 
284  data[lines.at(j)->cursorPos]=0;
285  aftercursor=&data[lines.at(j)->cursorPos+1];
286 //1stbit
287  cairo_show_text(this->cr,data);
288  cairo_text_extents (this->cr,data,&partextents);
289  cairo_text_extents (this->cr,undercurs,&charextents);
290  cairo_set_source_rgba(this->cr,this->cursorColour.RGBAColour.r,this->cursorColour.RGBAColour.g,this->cursorColour.RGBAColour.b,this->cursorColour.RGBAColour.a);
291 
292  cairo_rectangle(this->cr,partextents.x_advance+0.5,yoffset-this->fontExtents.ascent,charextents.x_advance-0.5,this->fontExtents.ascent+this->fontExtents.descent);
293  cairo_fill(this->cr);
294 //secondbit
295  cairo_move_to(this->cr,partextents.x_advance,yoffset);
296  cairo_set_source_rgba(this->cr,1.0,1.0,1.0,1.0);
297  cairo_show_text(this->cr,undercurs);
298 //3rdbit
299  cairo_set_source_rgba(this->cr,0.0,0.0,0.0,1.0);
300  if(lines.at(j)->cursorPos<strlen(lines.at(j)->line))
301  cairo_show_text(this->cr,aftercursor);
302 
303  free(data);
304  }
305  else
306  {
307  cairo_set_source_rgba(this->cr,0.0,0.0,0.0,1.0);
308  cairo_show_text(this->cr,lines.at(j)->line);
309  }
310  }
311  cairo_restore(this->cr);
312 
313  free(buffer);
314 }

320 {
321  Window selectionOwner;
322  unsigned char *data=NULL;
323  Atom type;
324  int format,result;
325  unsigned long len=0,bytesLeft=0,dummy=0;
326  bool run=true;
327  XEvent event;
328 
329  selectionOwner=XGetSelectionOwner(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD));
330 
331  if(selectionOwner==this->wc->window)
332  {
333  this->LFSTK_setFormatedText(this->wc->clipBuffer.c_str(),false);
334  this->LFSTK_clearWindow();
335  return;
336  }
337 
338  if (selectionOwner!=None)
339  {
340  XConvertSelection(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->wc->LFSTK_getDnDAtom(XA_UTF8_STRING),this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->window,CurrentTime);
341  XFlush(this->display);
342 
343  while (run==true)
344  {
345  XNextEvent(this->display,&event);
346  switch(event.type)
347  {
348  case SelectionNotify:
349  if(event.xselection.requestor==this->window)
350  run=false;
351  break;
352  }
353  }
354 
355  XGetWindowProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),0,0,False,AnyPropertyType,&type,&format,&len,&bytesLeft,&data);
356  if(data!=NULL)
357  {
358  XFree(data);
359  data=NULL;
360  }
361 
362  if(bytesLeft>0)
363  {
364  result=XGetWindowProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),0,bytesLeft,False,AnyPropertyType,&type,&format,&len,&dummy,&data);
365  if (result==Success)
366  {
367  this->LFSTK_setFormatedText((const char*)data,false);
368  XFree(data);
369  }
370  }
371  XDeleteProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD));

383 bool LFSTK_multiLineEditClass::keyRelease(XKeyEvent *e)
384 {
385  char c[255];
386  KeySym keysym_return;
387  char *command;
388 
389  if(this->isActive==false)
390  {
391  this->LFSTK_clearWindow();
392  return(true);
393  }
394 
395  XLookupString(e,(char*)&c,255,&keysym_return,NULL);
396 
397  if(this->isFocused==false)
398  return(true);
399 
400  if(e->state==ControlMask)
401  {
402  if(keysym_return==XK_v)
403  this->getClip();
404 //TODO//
405  if(keysym_return==XK_c)
406  {
407  this->wc->clipBuffer=this->buffer;
408  XSetSelectionOwner(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->wc->window,CurrentTime);
409  }
410 
411  if(keysym_return==XK_Delete)
412  {
413  this->LFSTK_setBuffer("");
414  }
415  }
416  else
417  {
418  switch(keysym_return)
419  {
420  case XK_Tab:
421  this->buffer.insert(this->cursorPos,4,' ');
422  this->cursorPos+=4;
423 
424  break;
425  case XK_BackSpace:
426  if(this->cursorPos>0)
427  {
428  this->buffer.erase(this->cursorPos-1,1);
429  this->cursorPos--;
430  }
431  break;
432  case XK_Delete:
433  this->buffer.erase(this->cursorPos,1);
434  break;
435  case XK_Left:
436  if(this->cursorPos>0)
437  this->cursorPos--;
438  break;
439  case XK_Right:
440  if(this->cursorPos<this->buffer.length())
441  this->cursorPos++;
442  break;
443  case XK_End:
444  while((this->cursorPos<this->buffer.length()) && (this->buffer[this->cursorPos]!='\n'))
445  this->cursorPos++;
446  break;
447  case XK_Page_Down:
448  this->cursorPos=this->buffer.length();
449  break;
450  case XK_Home:
451  while((this->cursorPos>0) && (this->buffer[this->cursorPos]!='\n'))
452  this->cursorPos--;
453  if(this->cursorPos!=0)
454  this->cursorPos++;
455  break;
456  case XK_Page_Up:
457  this->cursorPos=0;
458  break;
459  case XK_Down:
460  if(this->cursorPos>this->buffer.length())
461  break;
462  do
463  this->cursorPos++;
464  while((this->cursorPos<this->buffer.length()) && (this->buffer[this->cursorPos]!='\n'));
465  if(this->cursorPos>this->buffer.length())
466  this->cursorPos=this->buffer.length();
467  break;
468  case XK_Up:
469  if(this->cursorPos==0)
470  break;
471  do
472  this->cursorPos--;
473  while((this->cursorPos>0) && (this->buffer[this->cursorPos]!='\n'));
474  break;
475  case XK_Select ... XK_Num_Lock:
476  case XK_F1 ... XK_R15:
477  break;
478  case XK_Return:
479  this->buffer.insert(this->cursorPos,1,'\n');
480  this->cursorPos++;
481  if(this->callback.pressCallback!=NULL)
482  return(this->callback.pressCallback(this,this->callback.userData));
483  break;
484 
485  default:
486  if(c[0]==0)
487  break;
488  this->buffer.insert(this->cursorPos,1,c[0]);
489  this->cursorPos++;
490  break;
491  }
492  }
493 
494  this->setDisplayLines();
495  this->LFSTK_clearWindow();
496  if(this->callback.releaseCallback!=NULL)
497  return(this->callback.releaseCallback(this,this->callback.userData));
498 
499  return(true);
500 }

534 void LFSTK_multiLineEditClass::setDisplayLines(void)
535 {
536  cairo_text_extents_t extents;
537  char data[4096]={0,};
538  int len=0;
539  geometryStruct geom;
540  lineStruct *newline;
541  int cursorpos=-1;
542  bool donecurs=false;
543 
544  this->LFSTK_getGeom(&geom);
545 
546  for (int i=0; i<lines.size(); i++)
547  {
548  free(lines.at(i)->line);
549  delete lines.at(i);
550  }
551  this->lines.clear();
552 
553  for(int j=0;j<this->buffer.length();j++)
554  {
555  if((this->cursorPos==j) && (this->isFocused==true))
556  {
557  cursorpos=len;
558  donecurs=true;
559  }
560 
561  data[len]=this->buffer.c_str()[j];
562  data[len+1]=0;
563 
564  cairo_select_font_face(this->cr,fontName,slant,weight);
565  cairo_set_font_size(this->cr,fontSize);
566  cairo_text_extents(this->cr,data,&extents);
567  if(((extents.x_advance)>geom.w) || (data[len]=='\n'))
568  {
569  if(data[len]!='\n')
570  j--;
571  data[len++]=0;
572  newline=new lineStruct;
573  asprintf(&newline->line,"%s",data);
574  newline->cursorPos=cursorpos;
575  newline->xpos=0;
576  newline->ypos=0;
577  newline->width=extents.width;
578  newline->height=extents.height;
579  lines.push_back(newline);
580  data[0]=0;
581  len=0;
582  cursorpos=-1;
583  }
584  else
585  {
586  len++;
587  data[len]=0;
588  }
589  }
590 //last line
591  if(strlen(data)>0)
592  {
593  newline=new lineStruct;
594  asprintf(&newline->line,"%s",data);
595  if((donecurs==false) && (this->isFocused==true))
596  {
597  cursorpos=strlen(newline->line);

599  }

601  newline->xpos=0;
602  newline->ypos=0;
603  newline->width=extents.width;
604  newline->height=extents.height;
605  lines.push_back(newline);
606  }
607 //special case last line is newline
608  if((donecurs==false) && (this->isFocused==true))
609  {
610  newline=new lineStruct;
611  asprintf(&newline->line," ");
612  newline->cursorPos=0;
613  lines.push_back(newline);
614  }
615  this->LFSTK_clearWindow();
