140 void LFSTK_scrollBarClass::setState(bool byvalue)
141 {
142  geometryStruct geom;
143  rectStruct rect;
144  int diff;
145 
146  this->thumb->LFSTK_getGeom(&geom);
147  this->thumb->LFSTK_getLimits(&rect);
148 
149  if(byvalue==true)
150  {
151  if(this->verticalBar==false)
152  {
153  geom.x=rect.x+(int)((this->value*this->scale)+0.5);
154  if(this->value<this->minScale)
155  {
156  geom.x=rect.x+1;
157  this->value=this->minScale;
158  }
159  if(this->value>this->maxScale)
160  {
161  geom.x=rect.w-geom.h;
162  this->value=this->maxScale;
163  }
164  if(geom.x==rect.x)
165  geom.x++;
166  }
167  else
168  {
169  geom.y=rect.y+(int)((this->value*this->scale)+0.5);
170  if(this->value<this->minScale)
171  {
172  geom.y=rect.y+1;
173  this->value=this->minScale;
174  }
175  if(this->value>this->maxScale)
176  {
177  geom.y=rect.h-geom.w;
178  this->value=this->maxScale;
179  }
180  if(geom.y==rect.y)
181  geom.y++;
182  }
183 
184  this->thumb->LFSTK_moveGadget(geom.x,geom.y);
185  }
186  else
187  {
188  if(this->verticalBar==false)
189  diff=geom.x-1-rect.x;
190  else
191  diff=geom.y-1-rect.y;
192 
193  this->value=(int)(double)((diff/this->scale)+0.5);
194 
195  if(this->value<this->minScale)
196  this->value=this->minScale;
197  if(this->value>this->maxScale)
198  this->value=this->maxScale;
199  return;
200  }
201 
202  this->LFSTK_clearWindow();
203 }

329 bool LFSTK_scrollBarClass::mouseUp(XButtonEvent *e)
330 {
331  geometryStruct geom;
332  rectStruct rect;
333 
334  this->mouseDownX=e->x;
335  this->mouseDownY=e->y;
336 
337  if((this->isActive==false) || (this->callback.ignoreCallback==true))
338  return(true);
339 
340  if(this->inWindow==true)
341  {
342  this->thumb->LFSTK_getGeom(&geom);
343  this->thumb->LFSTK_getLimits(&rect);
344 
345  switch(e->button)
346  {
347  case Button1:
348  if(this->verticalBar==false)
349  {
350  if((this->mouseDownX+geom.w)<geom.x-rect.y)
351  this->value-=this->pageScroll;
352  else
353  this->value+=this->pageScroll;
354  }
355  else
356  {
357  if((this->mouseDownY-geom.h)<geom.y-rect.y)
358  this->value-=this->pageScroll;
359  else
360  this->value+=this->pageScroll;
361  }
362  break;
363  case Button4:
364  this->value-=this->pageScroll;
365  break;
366  case Button5:
367  this->value+=this->pageScroll;
368  break;
369  }
370  this->setState(true);
371  }
372  return(true);
373 }
