235 void LFSTK_lineEditClass::drawLabel(void)
236 {
237  int startchar=0;
238  double yoffset=0;
239  cairo_text_extents_t partextents;
240  char *data;
241  char undercurs[2]={0,0};
242  char *aftercursor;
243  int maxchars;

245 
246  cairo_save(this->cr);
247  cairo_reset_clip(this->cr);
248  cairo_set_source_rgba(this->cr,this->colourNames[NORMALCOLOUR].RGBAColour.r,this->colourNames[NORMALCOLOUR].RGBAColour.g,this->colourNames[NORMALCOLOUR].RGBAColour.b,this->colourNames[NORMALCOLOUR].RGBAColour.a);
249  cairo_paint(this->cr);
250  cairo_restore(this->cr);
251 
252  cairo_save(this->cr);
253  cairo_select_font_face(this->cr,this->fontName,this->slant,this->weight);
254  cairo_set_font_size(this->cr,fontSize);
255  cairo_set_source_rgba(this->cr,0.0,0,0,1.0);
256 
257  if(this->LFSTK_getTextWidth("X")<=0)
258  maxchars=((this->gadgetGeom.w)/(int)((8)));
259  else
260  maxchars=(this->gadgetGeom.w/(int)(this->LFSTK_getTextRealWidth("X")+0.5))-1;
261 
262  if(maxchars>this->buffer.length())
263  maxchars=this->buffer.length();
264  startchar=this->offsetCurs;
265  yoffset=(this->gadgetDetails.gadgetGeom.h/2)-(this->textExtents.y_bearing/2);
266  cairo_move_to(this->cr,this->pad,yoffset);
267 
268  if(startchar<0)
269  startchar=0;
270 
271  asprintf(&data,"%s",this->buffer.substr(startchar,this->cursorPos-startchar).c_str());
272  undercurs[0]=this->buffer.c_str()[this->cursorPos];
273  if(undercurs[0]==0)
274  undercurs[0]=' ';
275 
276  if(this->cursorPos-startchar>=maxchars)
277  asprintf(&aftercursor,"");
278  else
279  {
280  if(this->cursorPos+1<this->buffer.length())
281  asprintf(&aftercursor,"%s",this->buffer.substr(this->cursorPos+1,(maxchars-this->visCursorPos)).c_str());
282  else
283  asprintf(&aftercursor,"");
284  }
285 //1stbit
286  cairo_show_text(this->cr,data);
287  cairo_text_extents(this->cr,data,&partextents);
288  cairo_text_extents (this->cr,undercurs,&charextents);
289 
290  if(this->isFocused==true)
291  {
292  cairo_set_source_rgba(this->cr,this->cursorColour.RGBAColour.r,this->cursorColour.RGBAColour.g,this->cursorColour.RGBAColour.b,this->cursorColour.RGBAColour.a);
293  cairo_rectangle(this->cr,partextents.x_advance+this->pad,yoffset-this->fontExtents.ascent,charextents.x_advance-0.5,this->fontExtents.ascent+this->fontExtents.descent);

296  }
297 //secondbit
298  cairo_move_to(this->cr,partextents.x_advance+this->pad,yoffset);
299  cairo_show_text(this->cr,undercurs);
300 //3rdbit

302  cairo_show_text(this->cr,aftercursor);
303

312 void LFSTK_lineEditClass::getClip(void)
313 {
314  Window selectionOwner;
315  unsigned char *data=NULL;
316  Atom type;
317  int format,result;
318  unsigned long len=0,bytesLeft=0,dummy=0;
319  bool run=true;
320  XEvent event;
321 
322  selectionOwner=XGetSelectionOwner(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD));
323 
324  if(selectionOwner==this->wc->window)
325  {
326  this->LFSTK_setFormatedText(this->wc->clipBuffer.c_str(),false);
327  this->LFSTK_clearWindow();
328  return;
329  }
330 
331  if (selectionOwner!=None)
332  {
333  XConvertSelection(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->wc->LFSTK_getDnDAtom(XA_UTF8_STRING),this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->window,CurrentTime);
334  XFlush(this->display);
335 
336  while (run==true)
337  {
338  XNextEvent(this->display,&event);
339  switch(event.type)
340  {
341  case SelectionNotify:
342  if(event.xselection.requestor==this->window)
343  run=false;
344  break;
345  }
346  }
347 
348  XGetWindowProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),0,0,False,AnyPropertyType,&type,&format,&len,&bytesLeft,&data);
349  if(data!=NULL)
350  {
351  XFree(data);
352  data=NULL;
353  }
354 
355  if(bytesLeft>0)
356  {
357  result=XGetWindowProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),0,bytesLeft,False,AnyPropertyType,&type,&format,&len,&dummy,&data);
358  if (result==Success)
359  {
360  this->LFSTK_setFormatedText((const char*)data,false);
361  XFree(data);
362  }
363  }
364  XDeleteProperty(this->display,this->window,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD));
365  }
366 }

410 bool LFSTK_lineEditClass::keyRelease(XKeyEvent *e)
411 {
412  char c[255];
413  char *command;
414 
415  if(this->isActive==false)
416  {
417  this->LFSTK_clearWindow();
418  return(true);
419  }
420 
421  XLookupString(e,(char*)&c,255,&keysym_return,NULL);
422 
423  if(this->isFocused==false)
424  return(true);
425 
426  this->state=e->state;
427  if(e->state==ControlMask)
428  {
429  if(keysym_return==XK_v)
430  this->getClip();
431 //TODO//
432  if(keysym_return==XK_c)
433  {
434  this->wc->clipBuffer=this->buffer;
435  XSetSelectionOwner(this->display,this->wc->LFSTK_getDnDAtom(XA_CLIPBOARD),this->wc->window,CurrentTime);
436  }
437 
438  if(keysym_return==XK_Delete)
439  {
440  this->LFSTK_setBuffer("");
441  }
442  }
443  else
444  {
445  switch(keysym_return)
446  {
447  case XK_Tab:
448  this->buffer.insert(this->cursorPos,8,' ');
449  this->cursorPos+=8;
450  break;
451  case XK_BackSpace:
452  if(this->cursorPos>0)
453  {
454  this->buffer.erase(this->cursorPos-1,1);
455  this->setOffsetcurs(-1);
456  }
457  break;
458  case XK_Delete:
459  if(this->cursorPos<this->buffer.length())
460  this->buffer.erase(this->cursorPos,1);
461  this->setOffsetcurs(0);
462  break;
463  case XK_Left:
464  this->setOffsetcurs(-1);
465  break;
466  case XK_Right:
467  this->setOffsetcurs(1);
468  break;
469  case XK_End:
470  case XK_Page_Down:
471  this->setOffsetcurs(1000);
472  this->setOffsetcurs(1);
473  break;
474  case XK_Home:
475  case XK_Page_Up:
476  this->offsetCurs=0;
477  this->cursorPos=0;
478  this->visCursorPos=0;
479 
480  break;
481  case XK_Down:
482  case XK_Up:
483  case XK_Select ... XK_Num_Lock:
484  case XK_F1 ... XK_R15:
485  break;
486  case XK_Return:
487  if(this->callback.pressCallback!=NULL)
488  return(this->callback.pressCallback(this,this->callback.userData));
489  break;
490 
491  default:
492  if(c[0]==0)
493  break;
494  this->buffer.insert(this->cursorPos,1,c[0]);
495  this->setOffsetcurs(1);
496  break;
497  }
498  }
499 
500  this->LFSTK_clearWindow();
501  if(this->callback.releaseCallback!=NULL)
502  return(this->callback.releaseCallback(this,this->callback.userData));
503 
504  return(true);
505 }

544

546  {
547  switch(txt[j])
548  {
549  case '\t':
550  formtxt.append(" ");
551  len+=8;
552  break;
553  case '\n':
554  case 0:
555  case '\r':
556  break;
557  default:

562  }
563 
564  if(replace==true)

639  this->cursorPos=this->buffer.length()-1;
