109  break;

537 void LFSTK_fileDialogClass::LFSTK_showFileDialog(void)
538 {
539  XEvent event;
540  unsigned lasttime=0;
541  unsigned lastdiritem=0;
542  unsigned lastfileitem=0;
543  geometryStruct geomdir;
544  geometryStruct geomfile;
545  pointStruct pt;
546  char *lastdir=NULL;
547 
548  this->apply=false;
549  if(this->dialog!=NULL)
550  {
551  this->dirEdit->LFSTK_setBuffer(this->currentDir);
552  this->LFSTK_setWorkingDir(this->currentDir);
553  this->dialog->LFSTK_showWindow();
554  this->dialog->LFSTK_setKeepAbove(true);
555  this->dialog->LFSTK_setTransientFor(this->wc->window);
556 
557  this->mainLoop=true;
558  while(this->mainLoop==true)
559  {
560  XNextEvent(this->dialog->display,&event);
561  mappedListener *ml=this->dialog->LFSTK_getMappedListener(event.xany.window);
562  if(ml!=NULL)
563  ml->function(ml->gadget,&event,ml->type);
564 
565  if(this->dialog->LFSTK_handleWindowEvents(&event)<0)
566  this->mainLoop=false;
567 
568  if(ml==NULL)
569  continue;
570 
571  if(ml->gadget==this->dirEdit)
572  {
573  switch(event.type)
574  {
575  char c[4];
576  case KeyRelease:
577  KeySym keysym_return;
578  XLookupString(&event.xkey,(char*)&c,255,&keysym_return,NULL);
579  if(keysym_return==XK_Return)
580  {
581  if(this->isADir(this->dirEdit->LFSTK_getCStr()))
582  {
583  this->apply=false;
584  this->LFSTK_setWorkingDir(this->dirEdit->LFSTK_getCStr());
585  continue;
586  }
587  }
588  break;
589  }
590  continue;
591  }
592 
593  switch(event.type)
594  {
595  case ButtonRelease:
596  if(ml->gadget==this->buttonHidden)
597  {
598  this->showHidden=!this->showHidden;
599  this->LFSTK_setWorkingDir(this->currentDir);
600  break;
601  }
602 
603  if(ml->gadget==this->buttonApply)
604  {
605  char *buf;
606  char *rp;
607 
608  asprintf(&buf,"%s/%s",this->dirEdit->LFSTK_getCStr(),this->fileListGadget->labelData[this->fileListGadget->LFSTK_getCurrentListItem()]->label);
609  if((this->isADir(buf)==true) && (this->dialogType==FILEDIALOG))
610  {
611  this->apply=false;
612  this->LFSTK_setWorkingDir(buf);
613  free(buf);
614  break;
615  }
616 
617  if(this->dialogType==FOLDERDIALOG)
618  {
619  this->apply=true;
620  this->mainLoop=false;
621  this->setFileData();
622  free(buf);
623  break;
624  }
625  this->apply=true;
626  this->setFileData();
627  this->mainLoop=false;
628  free(buf);
629  break;
630  }
631 
632  if(ml->gadget==this->buttonCancel)
633  {
634  this->mainLoop=false;
635  this->apply=false;
636  break;
637  }
638  break;
639  }
640  }
641  this->dialog->LFSTK_hideWindow();
642  }
643 }
