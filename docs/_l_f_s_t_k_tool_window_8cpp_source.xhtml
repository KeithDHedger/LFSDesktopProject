32 void LFSTK_toolWindowClass::windowClassInitCommon(windowInitStruct *wi)

37  Atom xa;
38  Atom xa_prop[3];
39  bool gotargb;
40  XVisualInfo visual_template;
41  XVisualInfo *visual_list=NULL;
42  int nxvisuals=0;
43 
44  this->display=wi->display;
45  if(this->display==NULL)
46  exit(1);
47 
48  this->depth=32;
49  this->setWindowGeom(wi->x,wi->y,wi->w,wi->h,WINDSETALL);
50 
51  this->fontString=NULL;

54 
55  this->screen=DefaultScreen(this->display);

57  this->rootWindow=DefaultRootWindow(this->display);
58  this->loadMonitorData();
59 
60  wa.win_gravity=NorthWestGravity;
61  wa.override_redirect=wi->overRide;
62  wm_delete_window=XInternAtom(this->display,"WM_DELETE_WINDOW",0);
63 
64  visual_template.screen=this->screen;
65  visual_list=XGetVisualInfo(this->display,0,&visual_template,&nxvisuals);
66  gotargb=false;
67  for(int i=0;i<nxvisuals;i++)
68  {
69  if(visual_list[i].depth==32 && (visual_list[i].red_mask==0xff0000 && visual_list[i].green_mask==0x00ff00 && visual_list[i].blue_mask==0x0000ff ))
70  {
71  this->visual=visual_list[i].visual;
72  this->depth=visual_list[i].depth;
73  gotargb=true;
74  break;
75  }
76  }
77  XFree(visual_list);
78 
79  this->cm=XCreateColormap(this->display,this->rootWindow,this->visual,AllocNone);
80  if(gotargb==true)
81  {
82  wa.colormap=this->cm;
83  wa.border_pixel=0;
84  wa.background_pixel=0;
85 
86  this->window=XCreateWindow(this->display,this->rootWindow,wi->x,wi->y,wi->w,wi->h,0,this->depth,InputOutput,this->visual,CWColormap | CWBorderPixel |CWWinGravity|CWOverrideRedirect,&wa);
87  }
88  else
89  {
90  this->window=XCreateWindow(this->display,this->rootWindow,wi->x,wi->y,wi->w,wi->h,0,CopyFromParent,InputOutput,CopyFromParent,CWWinGravity|CWOverrideRedirect,&wa);
91  }
92 
93  XSelectInput(this->display,this->window,SubstructureRedirectMask|StructureNotifyMask|ButtonPressMask | ButtonReleaseMask | ExposureMask|LeaveWindowMask|FocusChangeMask|SelectionClear|SelectionRequest);
94 
95  XSetWMProtocols(this->display,this->window,&wm_delete_window,1);
96  xa=XInternAtom(this->display,"_NET_WM_ALLOWED_ACTIONS",False);
97  xa_prop[0]=XInternAtom(this->display,"_NET_WM_STATE_STICKY",False);
98  xa_prop[1]=XInternAtom(this->display,"_NET_WM_STATE_ABOVE",False);
99  xa_prop[2]=XInternAtom(this->display,"_NET_WM_STATE_BELOW",False);
100  xa_prop[3]=XInternAtom(this->display,"_NET_WM_ACTION_CHANGE_DESKTOP",False);
101 
102  if(xa!=None)
103  XChangeProperty(this->display,this->window,xa,XA_ATOM,32,PropModeAppend,(unsigned char *)&xa_prop,4);
104 
105  this->LFSTK_setWindowType(wi->windowType);
106  this->windowName=strdup(wi->name);
107  XStoreName(this->display,this->window,this->windowName);
108  classHint.res_name=this->windowName;
109  classHint.res_class=(char*)"LFSToolKit";
110  XSetClassHint(this->display,this->window,&classHint);
111 
112  this->gc=XCreateGC(this->display,this->window,0,NULL);
113  this->LFSTK_setFontString((char*)DEFAULTFONT);
114  this->LFSTK_setDecorated(wi->decorated);
115  this->initWindow(wi->loadVars);
116 
117  if(this->globalLib->LFSTK_getUseTheme()==true)
118  this->LFSTK_setTile(this->globalLib->LFSTK_getGlobalString(-1,TYPEWINDOWTILE),-1);
119  else
120  this->useTile=false;
121 
122  this->userHome=getenv("HOME");
123  asprintf(&this->configDir,"%s/.config/LFS",this->userHome);
124  this->closeDisplayOnExit=wi->shutDisplayOnExit;
125 
126  this->globalLib->LFSTK_setCairoSurface(this->display,this->window,this->visual,&this->sfc,&this->cr,wi->w,wi->h);
127 
128  switch(wi->level)
129  {
130  case BELOWALL:
131  XLowerWindow(this->display,this->window);
132  break;
133  case ABOVEALL:
134  XRaiseWindow(this->display,this->window);
135  break;
136  }
137  this->gadgetMap.clear();
138 }
