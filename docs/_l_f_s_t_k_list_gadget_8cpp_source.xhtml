100 void LFSTK_listGadgetClass::LFSTK_setListFromFile(const char *filepath,bool includeempty)
101 {
102  FILE *file=NULL;
103  char *buffer;
104  size_t linelen=0;
105  ssize_t read=0;
106  int linecnt=0;
107  char *lines=NULL;
108  long userdata=0;
109 
110  this->freeList();
111  listLabelStruct ls;
112 
113  if(filepath!=NULL)
114  {
115  lines=this->wc->globalLib->LFSTK_oneLiner("wc -l %s",filepath);
116  linecnt=atoi(lines);
117  free(lines);
118  file=fopen(filepath,"r");
119  if(file!=NULL)
120  {
121  while(!feof(file))
122  {
123  buffer=NULL;
124  linelen=0;
125  read=getline(&buffer,&linelen,file);
126  if(read>0)
127  {
128  buffer[strlen(buffer)-1]=0;
129  ls.imageType=NOTHUMB;
130  if(includeempty==true)
131  {
132  ls.label=strdup(buffer);
133  ls.userData=(void*)userdata;
134  this->LFSTK_appendToList(ls);
135  }
136  else
137  {
138  if(strlen(buffer)>0)
139  {
140  ls.label=strdup(buffer);
141  ls.userData=(void*)userdata;
142  this->LFSTK_appendToList(ls);
143  }
144  }
145  }
146  free(buffer);
147  userdata++;
148  }
149  fclose(file);
150  this->LFSTK_updateList();
151  }
152  }
153 }

163 void LFSTK_listGadgetClass::LFSTK_updateList(void)
164 {
165  int orient=MENU;
166 
167  this->listOffset=0;
168  this->currentItem=0;
169 
170  if(this->listDataArray->size()>maxShowing)
171  this->scrollBar->LFSTK_setScale(0,this->listDataArray->size()-this->maxShowing);
172  else
173  this->scrollBar->LFSTK_setScale(0,0);
174  this->listCntNew=this->listDataArray->size();
175  for(int j=0;j<this->maxShowing;j++)
176  {
177  this->data[j].userData=j;
178  if(j<this->listCntNew)
179  {
180  this->labels[j]->LFSTK_setLabel(this->listDataArray->at(j).label);
181  this->labels[j]->LFSTK_setActive(true);
182  if(this->listDataArray->at(j).imageType==NOTHUMB)
183  this->labels[j]->LFSTK_setLabelGravity(LEFT);
184  else
185  this->labels[j]->LFSTK_setLabelGravity(MENU);
186  this->listDataArray->at(j).listPos=j;
187 
188  if(this->listDataArray->at(j).imageType==CAIROTHUMB)
189  this->labels[j]->LFSTK_setImageFromSurface(this->listDataArray->at(j).data.surface,MENU,true);
190 
191  if(this->listDataArray->at(j).imageType==FILETHUMB)
192  this->labels[j]->LFSTK_setImageFromPath(this->listDataArray->at(j).data.imagePath,MENU,true);
193  }
194  else
195  {
196  this->labels[j]->LFSTK_setLabel("");
197  this->labels[j]->LFSTK_setActive(false);
198  this->labels[j]->LFSTK_setImageFromPath(NULL,MENU,true);
199  }
200  this->labels[j]->gadgetDetails.colour=&this->labels[j]->colourNames[NORMALCOLOUR];
201  this->labels[j]->gadgetDetails.state=NORMALCOLOUR;
202  this->labels[j]->LFSTK_clearWindow();
203  }
204  this->scrollBar->LFSTK_setValue(0);
205  this->wc->LFSTK_clearWindow(true);
206  this->currentItem=0;
207  this->setNavSensitive();
208 }

234 
235  list->data[j].userData=j+list->listOffset;
236  }
237  }
238 
239  for(int j=0;j<list->maxShowing;j++)
