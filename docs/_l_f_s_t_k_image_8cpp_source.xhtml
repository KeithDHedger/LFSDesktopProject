61 void LFSTK_imageClass::LFSTK_clearWindow(void)
62 {
63  int xoffset;
64  int yoffset;
65 
66  if(this->useImage==false)
67  {
68  cairo_save(this->shapecr);
69  cairo_reset_clip(this->shapecr);
70  cairo_set_operator(this->shapecr,CAIRO_OPERATOR_CLEAR);
71  cairo_rectangle(this->shapecr,0,0,this->gadgetGeom.w,this->gadgetGeom.h);
72  cairo_fill(this->shapecr);
73  cairo_restore(this->shapecr);
74 
75  XShapeCombineMask(this->display,this->window,ShapeBounding,0,0,cairo_xlib_surface_get_drawable(shapesfc),ShapeSet);
76  XFlush(this->display);
77  XSync(this->display,false);
78  return;
79  }
80 
81  xoffset=(this->gadgetGeom.w/2)-(this->imageWidth/2);
82  yoffset=(this->gadgetGeom.h/2)-(this->imageHeight/2);
83 
84  cairo_save(this->cr);
85  cairo_set_source_rgba (this->cr,1.0,1.0,1.0,this->alpha);
86  cairo_rectangle(this->cr,0,0,this->gadgetGeom.w,this->gadgetGeom.h);
87  cairo_fill(this->cr);
88  cairo_restore(this->cr);
89 
90  cairo_save(this->shapecr);
91  cairo_reset_clip(this->shapecr);
92  cairo_set_source_surface(this->cr,this->cImage,xoffset,yoffset);
93  cairo_paint_with_alpha(this->cr,this->alpha);
94  cairo_set_operator(this->shapecr,CAIRO_OPERATOR_CLEAR);
95  cairo_rectangle(this->shapecr,0,0,this->gadgetGeom.w,this->gadgetGeom.h);
96  cairo_fill(this->shapecr);
97  cairo_set_operator(this->shapecr,CAIRO_OPERATOR_OVER);
98  cairo_mask_surface(this->shapecr,this->cImage,xoffset,yoffset);
99  cairo_restore(this->shapecr);
100 
101  XShapeCombineMask(this->display,this->window,ShapeBounding,0,0,cairo_xlib_surface_get_drawable(shapesfc),ShapeSet);
102  XFlush(this->display);
103  XSync(this->display,false);
104 }
